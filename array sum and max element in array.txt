#include <iostream>
using namespace std;

int fun (int arr[] , int sum ){
    for (int i=0 ; i<5 ; i++){
        sum+=arr[i];
    }
    return sum;
}
int maximmum(int arr[] ){
     int max = arr[0];
    for (int i=1 ; i<5 ; i++){
       if (max <arr[i]){
           max = arr[i];
       }
    }
    return max;
}

int main(){
    int arr[5];
    for (int i = 0 ; i<5 ; i++){
        cin>>arr[i];
    }
    
    for (int i = 0 ; i<5 ; i++){
        cout<<arr[i]<<" ";
    }
    cout<<endl;
    int sum1=0;
    int sum2 = fun(arr , sum1 );
    cout<<sum2<<endl;
    int max = maximmum(arr);
    cout<<max;
    return 0;
}
