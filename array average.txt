void avg(int arr[]){
    int total =0;
    double average =0;
    int lowest =arr[0] , highest = arr[0];
    int second_highest = arr[0];
    int lowest_index = -1 , highest_index = -1; 
    for (int i = 0 ; i<10 ; i++){
        total+=arr[i];
        if (lowest > arr[i] ){
            lowest = arr[i];
            lowest_index = i;
        }
        if (highest < arr[i] ){
            highest =arr[i];
            highest_index =i;
        }
    }
    for(int i = 1 ; i< 10 ; i++){
        if (second_highest <arr[i] && arr[i]!= highest ){
            second_highest = arr[i];
        }
    }
    
    average = total/10;
    cout<<average<<endl;
    cout<<lowest <<endl <<"lowest index is: "<<lowest_index<<endl;
    cout<<highest <<endl <<"highest index is: "<<highest_index<<endl;
    cout<<second_highest<<endl;
}
void bubble_sort(int arr[]){
    int temp ;
    for(int i = 0 ; i<10 ; i++){
        for(int j = i+1 ; j<10 ; j++){
            if (arr[i]>arr[j]){
                temp = arr[i];
                arr[i] = arr[j];
                arr[j] = temp;
            }
        }
    }
}
void reverse_arr(int arr[]) {
    int temp;
    for (int i = 0; i < 5; i++) {
        temp = arr[i];
        arr[i] = arr[9 - i];
        arr[9 - i] = temp;
    }
}
