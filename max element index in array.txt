#include<iostream>
using namespace std;
int main(){
	int size ;
	int counter=0;
	cout<<"enter the size of an array: ";
	cin>>size;
	double arr[size];
	for(int i=0; i<size ;i++){
		cout<<"enter the element "<<i+1<<" ";
		cin>>arr[i];
	}
	double max=arr[0];
		for(int i=0; i<size ; i++){
			if(arr[i]>max){
				max=arr[i];
			}
			counter= i+1;	
		} 
	cout<<"the index of max number is: "<<counter;
	return 0;
}
