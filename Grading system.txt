#include <iostream>
using namespace std;

int main() {
    int m ; 
    cout<<"enter the no of student: ";
    cin>>m;
    int roll_no , n , subject;
    double total , percentage; 
    string name;
    cout<<"enter the no of subject: ";
    cin>>n;
    for (int i = 0 ; i<m ; i++){
        cout<<"enter the roll no: ";
        cin>>roll_no;
        cout<<"enter your name: ";
        cin>>name;
        for(int j = 0 ; j<n ; j++){
            cout<<"enter the marks of subject "<<j+1<<": ";
            cin>>subject;
            total+=subject;
        }
        cout<<"your total marks is:"<<total<<endl;
        percentage = (total/(n*100))*100;
        cout<<"your total percentage is:"<<percentage<<"%"<<endl;
    }
    if (percentage >=80){
        cout<<"First Division";
    }
    else if (percentage >=70){
        cout<<"second Division";
    }
    else if (percentage >=60){
         cout<<"third Division";
    }
    else {
        cout<<"fourth Division";
    }
    return 0;
}
