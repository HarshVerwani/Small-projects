#include<iostream>
using namespace std;
    const int  max_size = 100;
int main(){
    int rows , cols ;
    cout<<"Enter no of rows: ";
    cin>>rows;
    cout<<"Enter no of cols: ";
    cin>>cols;

    int arrx1[max_size][max_size]={0} , arrx2[max_size][max_size]={0} , arrx3[max_size][max_size]={0};

    for (int i=0 ; i<rows ; i++){
        for(int j=0 ;j<cols ; j++){
            cout<<"Enter the ["<<i<<"]["<<j<<"] element of first array: ";
            cin>>arrx1[i][j];
        }
    }

    for (int i = 0; i < rows; i++) {
        for (int j = 0; j < cols; j++) {
            cout << "Enter the [" << i << "][" << j << "] element of second array: ";
            cin >> arrx2[i][j]; 
        }
    }

    for (int i=0 ; i<rows ;i++){
        for (int j=0 ;j<cols ; j++){
            for( int k =0 ; k<cols ; k++){
                arrx3[i][j] += (arrx1[i][k]*arrx2[k][j]);
            }
        }
    }

    for (int i=0 ; i<rows ; i++){
        for(int j=0 ;j<cols ; j++){
            cout<<arrx3[i][j]<<" ";
        }
        cout<<endl;
    }

}
